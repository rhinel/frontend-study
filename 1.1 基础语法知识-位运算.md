## 位运算应用

```javascript

// 编码函数，不需要原始数组
const encode = arr => arr.reduce((s, v) => s + v, 0)
// => undefined

// 解码函数，需要通过原始数组解码
const decode = (arr, value) => arr.filter(v => value & v)
// => undefined

const arrAll = [1, 2, 4, 8, 16, 32];
// => undefined

const arrChoose = [4, 32];
// => undefined

// 编码选择函数
encode(arrChoose);
// => 36

// 通过原始数组，解码上面的编码结果
decode(arrAll, 36)
// => (2) [4, 32]

```
